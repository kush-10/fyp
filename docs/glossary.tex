\begin{dictionary}{Glossary}
\item[$t_{prove}$] Proof generation time (seconds).
\item[$t_{verify}$] Proof verification time (seconds).
\item[$t_{total}$] End-to-end measured time, $t_{prove}+t_{verify}$.
\item[$C_{total}$] Total zkVM cycle count reported by the prover.
\item[$C_{user}$] User cycle count attributed to guest computation.
\item[$C_{paging}$] Cycle count attributed to paging overhead.
\item[$C_{reserved}$] Cycle count reserved by runtime/prover internals.
\item[$\Delta\%$] Percentage change between baseline and optimized runs.
\item[$N_{sbox}$] Number of LowMC S-box instances per round (49).
\item[$B$] LowMC block size (256 bits in this implementation).
\end{dictionary}
